# 企业伴身智能平台 - 详细设计文档

## 第十部分：前端UI与交互设计规范

---

## 1. 设计原则与目标

### 1.1 设计原则

| 原则 | 说明 | 实践指南 |
|------|------|----------|
| 简洁高效 | 界面简洁，操作高效 | 减少视觉干扰，优化操作路径 |
| 一致性 | 视觉和交互风格统一 | 使用统一的组件库和设计规范 |
| 可用性 | 易学易用 | 符合用户习惯，提供明确引导 |
| 可访问性 | 无障碍访问 | 支持键盘导航，色彩对比度达标 |
| 响应式 | 多端适配 | 适配PC、平板、手机等设备 |
| 反馈及时 | 操作有反馈 | 实时状态反馈，错误提示明确 |

### 1.2 设计目标

- **字节跳动风格**：参考飞书等产品的设计语言
- **现代化审美**：简洁、清爽、专业的视觉风格
- **高效办公**：提升工作效率，减少操作成本
- **智能交互**：AI赋能，智能推荐和辅助

// API请求封装
export const userApi = {
  // 获取用户列表
  getList(params: UserListParams): Promise<PageResult<User>> {
    return request.get('/api/v1/users', { params });
  },
  
  // 获取用户详情
  getDetail(userId: number): Promise<User> {
    return request.get(`/api/v1/users/${userId}`);
  },
  
  // 创建用户
  create(data: CreateUserDTO): Promise<User> {
    return request.post('/api/v1/users', data);
  }
};

// 组合式函数
export function useUserList() {
  const loading = ref(false);
  const list = ref<User[]>([]);
  const total = ref(0);
  
  const fetchList = async (params: UserListParams) => {
    loading.value = true;
    try {
      const res = await userApi.getList(params);
      list.value = res.items;
      total.value = res.total;
    } catch (error) {
      message.error('获取用户列表失败');
    } finally {
      loading.value = false;
    }
  };
  
  return {
    loading,
    list,
    total,
    fetchList
  };
}

// 组件定义

  deptId?: number;

  select: [user: User];
}>();

onMounted(() => {
  fetchList({ deptId: props.deptId, page: 1, size: 20 });

</script>
