#### 2.2.4 权限校验流程

```java
// 权限校验注解使用
@RequiresPermission("user:user:list")
@GetMapping("/users")
public R<List<User>> listUsers() {
    return R.ok(userService.selectUserList());
}

// 权限校验拦截器
@Component
public class PermissionInterceptor implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, 
                            Object handler) throws Exception {
        // 1. 获取当前用户
        LoginUser loginUser = SecurityUtils.getLoginUser();
        
        // 2. 获取接口所需权限
        RequiresPermission annotation = getAnnotation(handler);
        if (annotation == null) {
            return true; // 无权限要求
        }
        
        // 3. 校验权限
        String[] permissions = annotation.value();
        if (!hasPermission(loginUser, permissions)) {
            throw new ForbiddenException("无权限访问");
        }
        
        return true;
    }
    
    private boolean hasPermission(LoginUser user, String[] required) {
        Set<String> userPermissions = user.getPermissions();
        for (String permission : required) {
            if (matchPermission(userPermissions, permission)) {
                return true;
            }
        }
        return false;
    }
    
    private boolean matchPermission(Set<String> userPerms, String required) {
        // 支持通配符匹配: user:* 可匹配 user:user:list
        for (String userPerm : userPerms) {
            if (userPerm.equals("*:*:*") || 
                matchWildcard(userPerm, required)) {
                return true;
            }
        }
        return false;
    }
}
```

#### 3.2.2 存储加密

| 数据类型 | 加密算法 | 密钥管理 | 说明 |
|----------|----------|----------|------|
| 用户密码 | bcrypt | 不可逆 | 12轮加盐哈希 |

```java
// 加密工具类
@Component
public class CryptoUtils {
    
    @Autowired
    private KmsClient kmsClient;
    
    // AES-256-GCM加密
    public String encryptAES(String plaintext, String keyId) {
        // 从KMS获取数据密钥
        DataKey dataKey = kmsClient.generateDataKey(keyId);
        
        // 加密
        Cipher cipher = Cipher.getInstance("AES/GCM/NoPadding");
        byte[] iv = generateRandomIV(12);
        GCMParameterSpec spec = new GCMParameterSpec(128, iv);
        cipher.init(Cipher.ENCRYPT_MODE, dataKey.getPlaintextKey(), spec);
        byte[] ciphertext = cipher.doFinal(plaintext.getBytes(StandardCharsets.UTF_8));
        
        // 格式: base64(iv) + "." + base64(encryptedDataKey) + "." + base64(ciphertext)
        return Base64.encode(iv) + "." + 
               Base64.encode(dataKey.getEncryptedKey()) + "." + 
               Base64.encode(ciphertext);
    }
    
    // AES-256-GCM解密
    public String decryptAES(String encrypted, String keyId) {
        String[] parts = encrypted.split("\\.");
        byte[] iv = Base64.decode(parts[0]);
        byte[] encryptedDataKey = Base64.decode(parts[1]);
        byte[] ciphertext = Base64.decode(parts[2]);
        
        // 从KMS解密数据密钥
        byte[] plaintextKey = kmsClient.decrypt(encryptedDataKey, keyId);
        
        // 解密
        Cipher cipher = Cipher.getInstance("AES/GCM/NoPadding");
        GCMParameterSpec spec = new GCMParameterSpec(128, iv);
        SecretKeySpec keySpec = new SecretKeySpec(plaintextKey, "AES");
        cipher.init(Cipher.DECRYPT_MODE, keySpec, spec);
        
        return new String(cipher.doFinal(ciphertext), StandardCharsets.UTF_8);
    }
    
    // 手机号哈希（用于查询）
    public String hashMobile(String mobile) {
        return DigestUtils.sha256Hex(mobile + getSalt());
    }
}
```

### 3.3 数据脱敏规则

```java
// 数据脱敏注解
@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.FIELD)
public @interface Sensitive {
    SensitiveType type();
}

// 脱敏类型
public enum SensitiveType {
    MOBILE,         // 手机号: 138****1234
    ID_CARD,        // 身份证: 110***********1234
    BANK_CARD,      // 银行卡: **** **** **** 1234
    EMAIL,          // 邮箱: z***@example.com
    NAME,           // 姓名: 张*
    ADDRESS,        // 地址: 北京市海淀区***
    PASSWORD,       // 密码: ******
    CUSTOM          // 自定义规则
}

// 脱敏处理器
@Component
public class SensitiveDataHandler {
    
    public String desensitize(String value, SensitiveType type) {
        if (StringUtils.isEmpty(value)) {
            return value;
        }
        
        switch (type) {
            case MOBILE:
                return value.replaceAll("(\\d{3})\\d{4}(\\d{4})", "$1****$2");
            case ID_CARD:
                return value.replaceAll("(\\d{3})\\d{11}(\\d{4})", "$1***********$2");
            case BANK_CARD:
                return "**** **** **** " + value.substring(value.length() - 4);
            case EMAIL:
                int atIndex = value.indexOf("@");
                if (atIndex > 1) {
                    return value.charAt(0) + "***" + value.substring(atIndex);
                }
                return "***" + value.substring(atIndex);
            case NAME:
                if (value.length() == 2) {
                    return value.charAt(0) + "*";
                } else if (value.length() > 2) {
                    return value.charAt(0) + "*".repeat(value.length() - 2) + 
                           value.charAt(value.length() - 1);
                }
                return "*";
            case ADDRESS:
                return value.substring(0, Math.min(6, value.length())) + "***";
            case PASSWORD:
                return "******";
            default:
                return value;
        }
    }
}

// 使用示例

    private String mobile;
    
    @Sensitive(type = SensitiveType.EMAIL)
    private String email;
    
    @Sensitive(type = SensitiveType.NAME)
    private String realName;
}
```

```java
// API安全配置
@Configuration
public class ApiSecurityConfig {
    
    // 1. 请求签名验证
    @Bean
    public SignatureFilter signatureFilter() {
        return new SignatureFilter() {
            @Override
            protected boolean verify(HttpServletRequest request) {
                String timestamp = request.getHeader("X-Timestamp");
                String nonce = request.getHeader("X-Nonce");
                String signature = request.getHeader("X-Signature");
                
                // 验证时间戳（5分钟有效）
                if (System.currentTimeMillis() - Long.parseLong(timestamp) > 300000) {
                    return false;
                }
                
                // 验证nonce（防重放）
                if (isNonceUsed(nonce)) {
                    return false;
                }
                
                // 验证签名
                String content = timestamp + nonce + getBodyMd5(request);
                String expected = HmacUtils.hmacSha256Hex(content, getSecretKey());
                
                return signature.equals(expected);
            }
        };
    }
    
    // 2. 请求限流
    @Bean
    public RateLimiter rateLimiter() {
        return new SlidingWindowRateLimiter() {
            @Override
            protected RateLimitConfig getConfig(HttpServletRequest request) {
                String path = request.getRequestURI();
                String userId = getCurrentUserId();
                
                // 不同接口不同限流配置
                if (path.startsWith("/api/v1/auth")) {
                    return new RateLimitConfig(10, 60, "ip:" + getClientIp(request));
                } else {
                    return new RateLimitConfig(300, 60, "user:" + userId);
                }
            }
        };
    }
    
    // 3. 参数校验
    @Bean
    public ParameterValidator parameterValidator() {
        return new ParameterValidator() {
            @Override
            protected void validate(Object param, Errors errors) {
                // XSS过滤
                filterXss(param);
                // SQL注入检测
                checkSqlInjection(param);
                // 长度限制
                checkLength(param);
            }
        };
    }
}
```

```java
// 异常检测服务
@Service
public class AbnormalDetectionService {
    
    // 异常检测规则
    private static final Map<String, DetectionRule> RULES = Map.of(
        // 登录异常规则
        "LOGIN_LOCATION", new DetectionRule(
            "异地登录检测",
            (ctx) -> {
                Location current = ctx.getCurrentLocation();
                Location last = ctx.getLastLoginLocation();
                double distance = calculateDistance(current, last);
                int timeDiff = ctx.getTimeDiffMinutes();
                
                // 1小时内距离超过500km判定为异常
                return distance > 500 && timeDiff < 60;
            },
            RiskLevel.HIGH
        ),
        
        "LOGIN_TIME", new DetectionRule(
            "异常时间登录",
            (ctx) -> {
                int hour = ctx.getLoginHour();
                return hour >= 0 && hour <= 5; // 凌晨登录
            },
            RiskLevel.MEDIUM
        ),
        
        "LOGIN_DEVICE", new DetectionRule(
            "陌生设备登录",
            (ctx) -> {
                return !ctx.isTrustedDevice();
            },
            RiskLevel.MEDIUM
        ),
        
        // 操作异常规则
        "DATA_EXPORT_FREQ", new DetectionRule(
            "频繁数据导出",
            (ctx) -> {
                int exportCount = ctx.getExportCountInHour();
                return exportCount > 10;
            },
            RiskLevel.HIGH
        ),
        
        "SENSITIVE_ACCESS", new DetectionRule(
            "敏感数据批量访问",
            (ctx) -> {
                int accessCount = ctx.getSensitiveAccessCountInMinute();
                return accessCount > 100;
            },
            RiskLevel.CRITICAL
        ),
        
        "PERMISSION_ESCALATION", new DetectionRule(
            "权限提升尝试",
            (ctx) -> {
                return ctx.isAccessingUnauthorizedResource();
            },
            RiskLevel.CRITICAL
        )
    );
    
    // 执行检测
    public List<AbnormalEvent> detect(AuditContext context) {
        List<AbnormalEvent> events = new ArrayList<>();
        
        for (Map.Entry<String, DetectionRule> entry : RULES.entrySet()) {
            DetectionRule rule = entry.getValue();
            if (rule.matches(context)) {
                AbnormalEvent event = new AbnormalEvent();
                event.setRuleCode(entry.getKey());
                event.setRuleName(rule.getName());
                event.setRiskLevel(rule.getRiskLevel());
                event.setContext(context);
                events.add(event);
                
                // 高风险事件实时告警
                if (rule.getRiskLevel().getValue() >= RiskLevel.HIGH.getValue()) {
                    alertService.sendAlert(event);
                }
            }
        }
        
        return events;
    }
}
```

```java
@Component
public class PerformanceMonitorAspect {
    
    @Around("@annotation(monitoredApi)")
    public Object monitor(ProceedingJoinPoint pjp, MonitoredApi monitoredApi) throws Throwable {
        String apiName = monitoredApi.name();
        long startTime = System.currentTimeMillis();
        boolean success = false;
        
        try {
            Object result = pjp.proceed();
            success = true;
            return result;
        } finally {
            long duration = System.currentTimeMillis() - startTime;
            
            // 记录指标
            metricsRecorder.record(apiName, duration, success);
            
            // 慢接口告警

                alertService.alertSlowApi(apiName, duration);
            }
            
            // 输出日志
            if (duration > 500) {
                log.warn("Slow API: {} took {}ms", apiName, duration);
            }
        }
    }
}

@Component
public class MetricsRecorder {
    
    private final MeterRegistry meterRegistry;
    
    public void record(String apiName, long duration, boolean success) {
        // 计数器
        Counter.builder("api.requests.total")
            .tag("api", apiName)
            .tag("status", success ? "success" : "failure")
            .register(meterRegistry)
            .increment();

            .tag("api", apiName)

            .record(duration, TimeUnit.MILLISECONDS);
            
        // 百分位数
        DistributionSummary.builder("api.response.latency")
            .tag("api", apiName)
            .publishPercentiles(0.5, 0.95, 0.99)
```

### 8.1 备份策略

| 数据类型 | 备份方式 | 备份频率 | 保留期限 | 恢复目标 |
